Дана JSON строка '["Коля", "Вася", "Петя"]'. Преобразуйте ее в массив JavaScript и выведите на экран элемент "Петя"


var name = '["Коля", "Вася", "Петя"]';
name = JSON.parse(name);
alert(name[2]);
